    <h1 class="text-2xl font-bold mb-6">Today's Session</h1>

    <div id="workout-list">Loading your lifts...</div>


    <script>

        const API_URL = "https://script.google.com/macros/s/AKfycbwGBgAJW4-MKmF3Xo-p5cntvu0VDzU4qH2xEP66hUHMdva2TBfjiC6THneUbSpKAMqx/exec";


        async function loadWorkout() {

            const response = await fetch(API_URL);

            const data = await response.json();

            const container = document.getElementById('workout-list');

            container.innerHTML = '';


            data.forEach(item => {

                const div = document.createElement('div');

                div.className = 'card';

                div.innerHTML = `

                    <div class="flex justify-between items-center mb-3">

                        <span class="font-bold text-lg text-blue-400">${item.exercise}</span>

                        <span class="text-gray-400">${item.sets} x ${item.reps}</span>

                    </div>

                    <div class="flex gap-4">

                        <div><label class="block text-xs text-gray-500">Weight</label><input type="number" id="w-${item.row}" value="${item.weight}"></div>

                        <div><label class="block text-xs text-gray-500">RPE</label><input type="number" id="r-${item.row}" value="${item.rpe}"></div>

                        <button onclick="saveRow(${item.row})" class="mt-4">Save</button>

                    </div>

                `;

                container.appendChild(div);

            });

        }


        async function saveRow(row) {

            const weight = document.getElementById(`w-${row}`).value;

            const rpe = document.getElementById(`r-${row}`).value;

            

            const btn = event.target;

            btn.innerText = "Saving...";

            btn.style.background = "#555";


            await fetch(API_URL, {

                method: "POST",

                body: JSON.stringify({ row, weight, rpe })

            });


            btn.innerText = "Saved!";

            btn.style.background = "#28a745";

            setTimeout(() => { btn.innerText = "Save"; btn.style.background = "#007aff"; }, 2000);

        }


        loadWorkout();

    </script>
